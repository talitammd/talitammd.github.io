path路径模块

单纯针对路径的操作，对文件没有关联，因此即使文件不存在也没关系

  • path.basename(dir) ：返回路径的最后一部分
  • path.dirname(dir) ：返回路径除了最后一部分的部分
  • path.format(dir) ：将对象解析为路径并返回
  • path.parse(dir) ：将路径解析为对象并返回
  • path.resolve(dir) ：将一个路径解析为绝对路径并返回
  • path.join(dir)或path.join([...paths]) ：将路径拼接并返回
  • path.extname(path)：返回路径的扩展名

两个全局变量：

  • __dirname ：当前文件的绝对路径
  • __filename ：当前文件的绝对路径加文件名
let path = require('path')
let demo = './src/a/b/c.js'
console.log(path.basename(demo))
// 输出c.js
console.log(path.dirname(demo))
// 输出./src/a/b/c.js
console.log(path.parse(demo))
//输出{ root: '', dir: './src/a/b', base: 'c.js', ext: '.js', name: 'c' }
console.log(path.format({ root: '', dir: './src/a/b', base: 'c.js', ext: '.js', name: 'c' }))
console.log(__dirname)
// 输出/Users/v/nodeStudy
console.log(__filename)
// 输出/Users/v/nodeStudy/code.js
console.log(path.resolve(__dirname,'./do'))
// 输出/Users/v/nodeStudy/do
console.log(path.resolve('../','./u'))
// 输出/Users/v/u
console.log(path.join('./love','/me'))
// 输出love/me

  • argv：启动node时的命令行参数，是用户自己定义的参数，写在末尾，比如webpack打包的时候，输入的—help 就是通过argv来读取的

  • execArgv：node命令后的直接参数，是node自带的参数，写在node后面

  • stdout和stdin：屏幕输出和输入对象

方法：

  • cwd() ：获取当前进程工作目录，该方法和__dirname 看似一样，但是当你cd 到上级目录后，__dirname 随之改变，但是工作进程目录是不变的，你的js文件在哪个目录，它就永远是哪个目录

  • exit() ：退出进程

  • memoryUsage() ：获取内存使用情况

示例1：获取和修改环境变量

// 获取环境变量  
const envVar = process.env.LOGNAME;  
console.log(envVar);  
  
// 修改环境变量（注意：这不会修改父进程或操作系统的环境变量，只会影响当前Node.js进程）  
process.env.LOGNAME = 'new value';  
console.log(process.env.LOGNAME); // 输出 'new value'

示例2：监听进程事件

// 监听退出事件  
process.on('exit', (code) => {  
  console.log(`About to exit with code: ${code}`);  
});  
  
// 当Node.js接收到SIGTERM或SIGINT信号时，会触发'exit'事件  
process.kill(process.pid, 'SIGTERM'); // 这将导致进程退出并触发'exit'事件  
  
// 监听未捕获的异常  
process.on('uncaughtException', (err) => {  
  console.error('Uncaught Exception:', err);  
  process.exit(1); // 强制退出进程  
});  
  
// 抛出一个未捕获的异常来触发'uncaughtException'事件  
setTimeout(() => {  
  throw new Error('This is an uncaught exception');  
}, 100);

示例3：获取进程信息

// 获取当前进程的PID  
console.log('Process ID:', process.pid);  
  
// 获取当前进程的标题（可以通过process.title修改）  
console.log('Process title:', process.title);  
  
// 获取当前Node.js的版本  
console.log('Node.js version:', process.version);  
  
// 获取当前Node.js平台的架构  
console.log('Platform architecture:', process.arch);  
  
// 获取当前操作系统平台  
console.log('Operating system platform:', process.platform);

示例4：发送信号给进程

// 发送信号给当前进程（这通常不是一个好实践，但只是为了演示）  
process.kill(process.pid, 'SIGTERM'); // 这将导致进程退出  
  
// 在实际中，你可能会发送信号给子进程或其他进程  
// 例如，使用child_process模块创建子进程后，可以使用process.kill发送信号

示例5：获取工作台输入

process.stdout.write('请输入一个数字')
process.stdin.on('data',res=>{
    console.log(res.toString())
        process.exit()
})

使用on对stdin进行监听，一旦获取到data，就执行回调函数，以实现和控制台的交互。比如vue-cli打包的时候，一直通过控制台和用户交互，用的就是这种方式

util工具模块

该模块提供了许多实用的工具函数，使得开发者能够更便捷地进行各种常见的操作

类型判断：

  • util.isArray(object)
  • util.isBoolean(object)
  • util.isNull(object)
  • util.isNullOrUndefined(object)
  • util.isNumber(object)
  • util.isString(object)
  • util.isSymbol(object)
  • util.isUndefined(object)
  • util.isObject(object)：判断一个对象是否是一个对象（不是数组、null、undefined、字符串、数字、布尔值或 Symbol）
  • util.isPrimitive(object)：判断一个对象是否是原始类型（字符串、数字、布尔值、Symbol 或 null）
  • util.isFunction(object)：判断一个对象是否是一个函数
  • util.isRegExp(object)：判断一个对象是否是一个正则表达式
  • util.isError(object)：判断一个对象是否是一个 Error 对象
  • util.types：这是一个对象，提供了各种用于检查对象类型的函数，如 util.types.isBigInt64Array(object)、util.types.isBigUint64Array(object) 等。

回调转换：

  • util.promisify(original)：把异步函数包装为promise对象形式
let util = require('util')
let fs = require('fs')

fs.readFile('./test.txt', (error, content) => {
    console.log(content.toString())
})
//转换为Promise形式以后，可以使用then API了
const promiseReadFile = util.promisify(fs.readFile)
promiseReadFile('./test.txt').then(res => {
    console.log(res.toString())
})
  • util.callbackify(asyncFn)：将一个返回值为promise的函数改为回调
let util = require('util')
function f1() {
    return new Promise(
        res => {
            setTimeout(() => {
                res(20)
                //回调产生一个值10
            }, 2000)
        }
    )
}
const callbackf1 = util.callbackify(f1)
callbackf1(function (num) {
    console.log(num)
})

fs文件模块

fs模块是Node.js中用于与文件系统进行交互的核心模块之一

常用文件操作

  1. 读取文件：readFile

    fs.readFile('example.txt', 'utf8', (err, data) => {  
  if (err) throw err;  
  console.log(data);  
});

  2. 写入并创建文件，如果文件已存在，已存在的文件将被覆盖：writeFile

    fs.writeFile('example.txt', 'Hello, World!', 'utf8', (err) => {  
  if (err) throw err;  
  console.log('The file has been saved!');  
});

  3. 向文件末尾追加内容，如果文件不存在，它将被创建：appendlink

    fs.appendFile('example.txt', 'Appended text', 'utf8', (err) => {  
  if (err) throw err;  
  console.log('The "data to append" was appended to file!');  
});

  4. 删除文件：unlink

  5. 移动/重命名文件：rename

  6. 拷贝文件：copyFile

总结：

  • 所有文件方法都是异步操作，以一个回调函数作为最后的参数，路径作为第一个参数通常
  • 所有文件方法都有一个同步版本，就是在原方法后加一个Sync，它不用回调函数作为参数，写入结果直接返回
  • 需要注意的是，如果既有同步的文件操作，又有异步的文件操作，事件循环机制会使同步操作先执行，而导致异步操作的结果可能对同步结果进行覆盖

常用文件夹操作

  1. 读取文件夹目录中的文件：readdir

    fs.readdir('./', (err, files) => {  
  files.forEach(file => {  
    console.log(file);  
  });  
});

  2. 创建空文件夹：mkdir

    fs.mkdir('./newDir', { recursive: true }, (err) => {  
  if (err) throw err;  
  console.log('Directory has been created!');  
});

  3. 删除（只有空文件夹可以删除）：rmdir

    fs.rmdir('./newDir', { recursive: true }, (err) => {  
  if (err) throw err;  
  console.log('Directory has been deleted!');  
});

  4. 拷贝和移动：没有直接api，必须手动遍历

    • 递归清空文件夹中的文件
    const fs = require('fs')
const ph = require('path')

function empty(path) {
    if (fs.existsSync(path)) {
        const dir = fs.readdirSync(path)
        dir.forEach(item => {
            const fullpath = ph.join(path, item)
            const _state = fs.statSync(fullpath)
            if (_state.isDirectory()) {
                empty(fullpath)
            } else {
                fs.unlinkSync(fullpath)
            }
        })
    }
}
empty('./testdir')
    • 递归拷贝文件夹
    const fs = require('fs')
const ph = require('path')

function copy(srcpath, targetpath) {
    if (!fs.existsSync(targetpath)) {
        fs.mkdirSync(targetpath)
    }
    if (fs.existsSync(srcpath)) {
        const dir = fs.readdirSync(srcpath)
        dir.forEach(item => {
            const srcfullpath = ph.join(srcpath, item)
            const targetfullpath = ph.join(targetpath, item)
            const state = fs.statSync(srcfullpath)
            if (state.isDirectory()) {
                copy(srcfullpath, targetfullpath)
            } else {
                fs.copyFileSync(srcfullpath,targetfullpath)
            }
        })
    }
}
copy('./testdir', './dir')

    注意：如果没使用同步函数并且没有加入回调函数作为参数，运行时就会报错

  5. 获取文件或目录的详细信息（如大小、创建时间等）：stat()

    fs.stat('example.txt', (err, stats) => {  
  console.log(stats.isFile()); // 检查是否为文件  
  console.log(stats.size); // 文件大小（字节）  
});

常用检测文件夹方法

  1. 检查文件或文件夹是否存在：exists
  2. 获取文件状态信息，判断是文件还是文件夹：stat

其他文件操作

  • watch：监听文件是否发生改变，一旦改变则出发回调。因为一直处于监听状态，所以程序不会退出
  • open：打开后可以进行自由操作（基本不用了）
    • read：自由读取
    • write：自由写入
  • writeStream和readStream：以流形式进行操作，创建写入流和读取流
const fs=require('fs')
fs.watch('./copy.txt',(err,filename)=>{
    console.log(filename+'文件发生改变')
    process.exit()
})

好用的第三方库

💡 *注意：用之前先安装*

fs-extra ：提供了一些操作文件的方法，如果开发过程中经常使用文件操作的话，最好引入这个包，功能比原生的fs模块更强大

  • copy ：可以拷贝文件，也可以拷贝文件夹
  • emptyDir ：递归清空文件夹
  • remove ：删除文件或者文件夹
  • move ：移动文件或者文件夹
  • outputFile ：向文件中添加内容，如果文件不存在，则自动创建
const fs = require('fs-extra')
fs.copySync('./dir','./newdir')
fs.emptyDirSync('./dir')
fs.removeSync('./newdir')
fs.moveSync('./testdir','./dir/testdir')
fs.outputFileSync('./newcopy.txt','love u')

compressing ：帮助制作文件压缩包和解压文件压缩包，因为文件夹是不能进行网络传输的，因此想要上传文件夹必须先压缩。这个包里的回调方法采用的是promise，而不是回调函数

  • zip.compressDir ：压缩文件夹
  • zip.uncompress ：解压压缩包
const cm = require('compressing')
cm.zip.compressDir('./dir','./dirzip.zip').then(res=>{
    console.log('压缩成功')
})
cm.zip.uncompress('./dirzip.zip','./zip').then(res=>{
    console.log('解压成功')
})
